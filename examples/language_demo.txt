# 中文示例 / Chinese Example

# 基本计算
工资 = 8000
房租 = 2000
餐费 = 30 * 25
结余 = 工资 - 房租 - 餐费

# 位运算
颜色 = 0xFF8040
红色 = (颜色 >> 16) & 0xFF
绿色 = (颜色 >> 8) & 0xFF
蓝色 = 颜色 & 0xFF
bitmap(颜色, 1)

# 字节序交换
网络数据 = 0x12345678
本地数据 = swap(网络数据)

# 日期/时间运算
today = Y20260411
deadline = today + D10
diff = Y20260411 - Y20260101

# 工作日计算
交付日 = workday(Y20260411, 10)
