# 位运算示例

# RGB 颜色提取
颜色 = 0xFF8040
R = (颜色 >> 16) & 0xFF
G = (颜色 >> 8) & 0xFF
B = 颜色 & 0xFF

# 查看位结构（大端字节序）
bitmap(颜色, 1)

# IP 地址解析
IP = 0xC0A80001
段1 = (IP >> 24) & 0xFF
段2 = (IP >> 16) & 0xFF
段3 = (IP >> 8) & 0xFF
段4 = IP & 0xFF

# swap() 字节序交换
网络数据 = 0x12345678
本地数据 = swap(网络数据)
bitmap(网络数据, 1)
bitmap(本地数据, 0)

# hex() + swap() 组合
hex(swap(网络数据))

# hex() 16进制显示
hex(255)
hex(65535)
